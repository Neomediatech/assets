GNU Wget 1.25.0 built on linux-musl.

-cares +digest -gpgme +https +ipv6 -iri +large-file -metalink -nls 
+ntlm +opie -psl +ssl/openssl 

Wgetrc: 
    /etc/wgetrc (system)
Compile: 
    clang -DHAVE_CONFIG_H -DSYSTEM_WGETRC="/etc/wgetrc" 
    -DLOCALEDIR="/opt/wget-1.25.0/share/locale" -I. -I../lib -I../lib 
    -DHAVE_LIBSSL -DNDEBUG -g -O2 
Link: 
    clang -DHAVE_LIBSSL -DNDEBUG -g -O2 -static -lpcre -lssl -lcrypto 
    -ldl -pthread -lz ../lib/libgnu.a
